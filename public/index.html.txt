    <script>
        let countdownInterval;
        let scheduledTime = 0;

        async function updateStatus() {
            try {
                const response = await fetch('/api/status');
                const data = await response.json();
                
                const statusEl = document.getElementById('status');
                const countdownEl = document.getElementById('countdown');
                const adminControls = document.getElementById('adminControls');
                const watchContainer = document.getElementById('watchContainer');
                const scheduleBtn = document.getElementById('scheduleBtn');
                const cancelBtn = document.getElementById('cancelBtn');
                
                if (data.released) {
                    statusEl.textContent = '🎉 VIDEO RELEASED!';
                    statusEl.className = 'status released';
                    countdownEl.style.display = 'none';
                    adminControls.style.display = 'none';
                    watchContainer.style.display = 'block';
                    clearInterval(countdownInterval);
                } else if (data.canceled) {
                    statusEl.textContent = '❌ Release Canceled';
                    statusEl.className = 'status canceled';
                    countdownEl.style.display = 'none';
                    adminControls.style.display = 'block';
                    scheduleBtn.style.display = 'inline-block';
                    cancelBtn.style.display = 'none';
                    watchContainer.style.display = 'none';
                    clearInterval(countdownInterval);
                } else if (data.timeLeft > 0) {
                    statusEl.textContent = `⏳ Video releases in:`;
                    statusEl.className = 'status';
                    countdownEl.style.display = 'block';
                    adminControls.style.display = 'block';
                    scheduleBtn.style.display = data.scheduledTime ? 'none' : 'inline-block';
                    cancelBtn.style.display = data.scheduledTime ? 'inline-block' : 'none';
                    watchContainer.style.display = 'none';
                    
                    updateCountdown(data.timeLeft);
                } else {
                    statusEl.textContent = 'No video scheduled';
                    statusEl.className = 'status';
                    countdownEl.style.display = 'none';
                    adminControls.style.display = 'block';
                    scheduleBtn.style.display = 'inline-block';
                    cancelBtn.style.display = 'none';
                    watchContainer.style.display = 'none';
                }
            } catch (error) {
                console.error('Error:', error);
                document.getElementById('status').textContent = 'Error loading status';
            }
        }

        function updateCountdown(timeLeft) {
            const hours = Math.floor(timeLeft / (1000 * 60 * 60));
            const minutes = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((timeLeft % (1000 * 60)) / 1000);
            
            document.getElementById('countdown').textContent = 
                `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
        }

        async function scheduleRelease() {
            const videoFile = document.getElementById('videoFile').files[0];
            if (!videoFile) {
                alert('Please select a video file first!');
                return;
            }

            const formData = new FormData();
            formData.append('video', videoFile);

            try {
                const response = await fetch('/api/schedule', {
                    method: 'POST',
                    body: JSON.stringify({ videoPath: videoFile.name }),
                    headers: { 'Content-Type': 'application/json' }
                });
                const result = await response.json();
                if (result.success) {
                    alert('✅ Video scheduled for release in 12 hours!');
                    // Note: In production, you'd actually upload the file here
                    updateStatus();
                }
            } catch (error) {
                alert('Error scheduling release');
            }
        }

        async function cancelRelease() {
            if (confirm('Are you sure you want to cancel the release?')) {
                try {
                    const response = await fetch('/api/cancel', { method: 'POST' });
                    const result = await response.json();
                    if (result.success) {
                        alert('✅ Release canceled!');
                        updateStatus();
                    }
                } catch (error) {
                    alert('Error canceling release');
                }
            }
        }

        async function watchVideo() {
            window.open('/video', '_blank');
        }

        // Auto-update every second
        setInterval(updateStatus, 1000);
        updateStatus(); // Initial load
    </script>
