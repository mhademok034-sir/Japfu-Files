export default function handler(req, res) {
    // MUST match the date in status.js exactly
    const RELEASE_DATE = new Date("2026-05-12T11:00:00").getTime();
    const now = Date.now();

    res.setHeader('Access-Control-Allow-Origin', '*');

    if (now >= RELEASE_DATE) {
        // REPLACE 'your-video-file.mp4' with the actual name of your file 
        // that is inside your /videos folder
        res.status(200).json({ 
            url: '/videos/your-video-file.mp4' 
        });
    } else {
        // If someone tries to access early, return an error
        res.status(403).json({ 
            error: "Access Denied: The vault is still locked." 
        });
    }
}
